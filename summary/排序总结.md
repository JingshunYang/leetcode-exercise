快速排序
```java
public class Quick {
  public static void sort(int[] a, int lo, int hi) {
    int j = partition(a, lo, hi);
    sort(a, lo, j - 1);
    sort(a, j + 1, hi);
  }
  private static int partition(int[] a, int lo, int hi) {
    int i = lo;
    int j = hi + 1;
    int v = a[lo];
    while (true) {
      while (v < a[++i]) {
        if (i == hi) break;
      }
      while (v > a[++j]) {
        if (j == lo) break;
      }
      if (i >= j) break;
      exch(a, i, j);
    }
    exch(a, lo, j);
    return j;
  }
  private static void exch(int[] a, int lo, int hi) {
    int temp = a[lo];
    a[lo] = a[hi];
    a[hi] = temp;
  }
}
```
