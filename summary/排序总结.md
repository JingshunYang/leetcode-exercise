快速排序
```java
public class Quick {
  public static void sort(int[] a, int lo, int hi) {
    int j = partition(a, lo, hi);
    sort(a, lo, j - 1);
    sort(a, j + 1, hi);
  }
  private static int partition(int[] a, int lo, int hi) {
    int i = lo;
    int j = hi + 1;
    int v = a[lo];
    while (true) {
      while (v < a[++i]) {
        if (i == hi) break;
      }
      while (v > a[++j]) {
        if (j == lo) break;
      }
      if (i >= j) break;
      exch(a, i, j);
    }
    exch(a, lo, j);
    return j;
  }
  private static void exch(int[] a, int lo, int hi) {
    int temp = a[lo];
    a[lo] = a[hi];
    a[hi] = temp;
  }
}
```

归并排序
```java
public class Merge {
  public static void sort(int[] a) {
    int[] aux = new int[a.length];
    sort(a, aux, 0, a.length - 1);
  }

  private static void sort(int[] a, int[] aux, int lo, int hi) {
    if (hi <= lo) return;
    int mid = lo + (hi - lo) / 2;
    sort(a, aux, lo, mid);
    sort(a, aux, mid + 1, hi);
    merge(a, aux, lo, mid, hi);
  }

  private static void merge(int[] a, int[] aux, int lo, int mid, int hi) {
    for (int k = lo; k < hi; k++) {
      aux[k] = a[k];
    }
    int i = lo, j = mid + 1;
    for (int k = lo; k < hi; k++) {
      if (i > mid) a[k] = aux[j++];
      else if (j > hi) a[k] = aux[i++];
      else if (aux[i] < aux[j]) a[k] = aux[i++];
      else a[k] = aux[j++];
    }
  }
}
```

```java
class Heap {
  public void sort(int[] pq) {
    build_heap(pq);
    for (int i = pq.length - 1; i >= 0; i--) {
      swap(pq, i, 0);
      heapify(pq, i, 0);
    }
  }
  public void build_heap(int[] pq) {
    int last_node = pq.length - 1;
    int last_parent = (last_node - 1) / 2;
    for(int i = last_parent; i >= 0; i--) {
      heapify(pq, pq.length, i);
    }
  }
  private void heapify(int[] pq, int size, int i) {
    int c1 = 2 * i + 1;
    int c2 = 2 * i + 2;
    int max = i;
    if (c1 < n && pq[c1] > pq[max]) {
      max = c1;
    }
    if (c2 < n && pq[c2] > pq[max]) {
      max = c2;
    }
    if (max != i) {
      swap(pq, i, max);
    }
  }
  private void swap(int[] pq, int m, int n) {
    int temp = pq[m];
    pq[m] = pq[n];
    pq[n] = temp;
  }
}
```
